indicator(title = 'TuTuTu', shorttitle = 'TuTuTu', overlay = true)

// =========== 1) Inputs ==============
rangePeriod       = input.int(15, 'Range Period', minval = 2, tooltip="Number of candles to calculate the range.")
atrPeriod         = input.int(20, 'ATR Period', minval = 1, tooltip="Period for calculating ATR indicator (volatility).")
rangeMultiplier   = input.float(0.9, 'Range Multiplier', minval = 0, step = 0.1, tooltip="Multiplier to adjust the ATR value.")

// --- Additional Filters ---
useADXFilter      = input.bool(true, 'ADX Filter', tooltip="Enable ADX filter.")
adx_threshold     = input.int(40, 'ADX Threshold', tooltip="ADX threshold to activate the filter.")
useRSIFilter      = input.bool(true, 'RSI Filter', tooltip="Enable RSI filter.")
rsi_length        = input.int(10, 'RSI Length', tooltip="Period for calculating RSI indicator.")
useVolumeFilter   = input.bool(true, 'Volume Filter', tooltip="Enable volume filter.")
volumeThreshold   = input.float(1.5, 'Volume Multiplier', minval = 0, step = 0.1, tooltip="Multiplier to compare trading volume with the average.")

// --- ADX Settings ---
adx_smoothing     = 14
di_length         = 14

// --- Filter Combination Mode ---
filterMode        = input.string(title = 'Filter Mode', defval = 'No Filter', options = ['ALL', 'ANY', '2-of-3', 'No Filter'], tooltip="Filter combination mode: ALL (all conditions), ANY (at least one), 2-of-3 (two out of three), No Filter (no filter).")

// --- TP/SL Settings ---
useExitSignals    = input.bool(true, 'Use Exit Signals?', tooltip="Enable exit signals (TP/SL).")
tp_multiplier_1   = input.float(2.0, 'TP Multiplier 1', tooltip="Multiplier for the first take profit target.")
tp_multiplier_2   = input.float(3.5, 'TP Multiplier 2', tooltip="Multiplier for the second take profit target.")
tp_multiplier_3   = input.float(6.0, 'TP Multiplier 3', tooltip="Multiplier for the third take profit target.")
sl_multiplier     = input.float(1.8, 'SL Multiplier', tooltip="Multiplier for stop loss.")
tpOffset          = input.int(5, 'TP/SL Offset', minval = 1, tooltip="Time offset for drawing TP/SL lines.")
tpLineLength      = input.int(30, 'TP/SL Line Length', minval = 1, tooltip="Length of drawn TP/SL lines.")

// --- Moving Average Type Selection ---
maTypeList        = input.string('EMA', 'Moving Average Type', options = ['EMA'], tooltip="Choose the type of moving average:EMA")
rangeBoxColor     = input.color(color.new(#7E57C2, 70), 'Range Box Color', tooltip="Color of the dynamic range box.")
buySignalColor    = input.color(color.new(#66BB6A, 70), 'Buy Signal Color', tooltip="Color of the buy signal.")
sellSignalColor   = input.color(color.new(#FF7043, 70), 'Sell Signal Color', tooltip="Color of the sell signal.")


// =========== 2) Functions and Basic Calculations ============
f_getMA(src, length, maType) =>
    maType == 'SMA' ? ta.sma(src, length) : maType == 'EMA' ? ta.ema(src, length) : ta.wma(src, length)

// Calculate average price and volatility
averagePrice    = f_getMA(close, rangePeriod, maTypeList)
priceVolatility = ta.atr(atrPeriod) * rangeMultiplier
volumeMA        = ta.sma(volume, rangePeriod)

// ADX Calculation Function
dirmov(len) =>
    up        = ta.change(high)
    down      = -ta.change(low)
    plusDM    = na(up) ? na : (up > down and up > 0 ? up : 0)
    minusDM   = na(down) ? na : (down > up and down > 0 ? down : 0)
    truerange = ta.rma(ta.tr, len)
    plus      = fixnan(100 * ta.rma(plusDM, len) / truerange)
    minus     = fixnan(100 * ta.rma(minusDM, len) / truerange)
    [plus, minus]

adxFunc(diLen, adxLen) =>
    [plus, minus] = dirmov(diLen)
    sum           = plus + minus
    100 * ta.rma(math.abs(plus - minus) / (sum == 0 ? 1 : sum), adxLen)

// Calculate ADX and +DI/-DI
[plusDI, minusDI] = dirmov(di_length)
adx_value = adxFunc(di_length, adx_smoothing)

// Calculate RSI
rsi_value = ta.rsi(close, rsi_length)




// =========== 3) Filter Conditions ============
adx_condition_buy  = not useADXFilter or (adx_value > adx_threshold and plusDI > minusDI)
adx_condition_sell = not useADXFilter or (adx_value > adx_threshold and minusDI > plusDI)
rsi_condition_buy  = not useRSIFilter or rsi_value < 40
rsi_condition_sell = not useRSIFilter or rsi_value > 60
volume_condition  = not useVolumeFilter or volume >= volumeThreshold * volumeMA

// Function to calculate number of met conditions with weights
f_calcScore(_adxCond, _rsiCond, _volCond) =>
    _score = 0.0
    if _adxCond
        _score += 2.0  
    if _rsiCond
        _score += 1.0
    if _volCond
        _score += 1.0
    _score

var float minNeeded = 4.0
if filterMode == 'ALL'
    minNeeded := 4.0  
else if filterMode == 'ANY'
    minNeeded := 1.0  
else if filterMode == '2-of-3'
    minNeeded := 3.0  
else if filterMode == 'No Filter'
    minNeeded := 0.0  

scoreBuy  = f_calcScore(adx_condition_buy, rsi_condition_buy, volume_condition)
scoreSell = f_calcScore(adx_condition_sell, rsi_condition_sell, volume_condition)

buyCondition  = scoreBuy >= minNeeded
sellCondition = scoreSell >= minNeeded


// =========== 4) 偵測動態範圍============
var box dynamicBox = na            
var bool breakoutOccurred = false  
var bool signalTriggered = false   
var string boxState = "Active"
var int startBar = na
var float upperBound = na
var float lowerBound = na
var float topSeries = na
var float bottomSeries = na

var box lastBuyBox  = na
var box lastSellBox = na

isInRange = true
for i = 0 to rangePeriod - 1
    if math.abs(close[i] - averagePrice) > priceVolatility
        isInRange := false
        break

var label buySignalLabel  = na
var label sellSignalLabel = na
var line tp_line1         = na
var line tp_line2         = na
var line tp_line3         = na
var line sl_line          = na
var label tp_label1       = na
var label tp_label2       = na
var label tp_label3       = na
var label sl_label        = na

if boxState == "Active" and not na(dynamicBox)
    topSeries := dynamicBox.get_top()
    bottomSeries := dynamicBox.get_bottom()
    box.set_right(dynamicBox, bar_index)

if isInRange and not isInRange[1]
    if not na(dynamicBox)
        box.delete(dynamicBox)
        dynamicBox := na
    if not na(buySignalLabel)
        label.delete(buySignalLabel)
        buySignalLabel := na
    if not na(sellSignalLabel)
        label.delete(sellSignalLabel)
        sellSignalLabel := na
    if not na(tp_line1)
        line.delete(tp_line1)
        tp_line1 := na
    if not na(tp_line2)
        line.delete(tp_line2)
        tp_line2 := na
    if not na(tp_line3)
        line.delete(tp_line3)
        tp_line3 := na
    if not na(sl_line)
        line.delete(sl_line)
        sl_line := na
    if not na(tp_label1)
        label.delete(tp_label1)
        tp_label1 := na
    if not na(tp_label2)
        label.delete(tp_label2)
        tp_label2 := na
    if not na(tp_label3)
        label.delete(tp_label3)
        tp_label3 := na
    if not na(sl_label)
        label.delete(sl_label)
        sl_label := na

    upperBound  := averagePrice + priceVolatility
    lowerBound  := averagePrice - priceVolatility
    startBar    := bar_index - rangePeriod
    dynamicBox  := box.new(left = startBar, top = upperBound, right = bar_index, bottom = lowerBound, border_color = color.new(color.purple, 0), bgcolor = rangeBoxColor)
    signalTriggered := false
    breakoutOccurred := false
    boxState := "Active"

if boxState == "Active" and not na(dynamicBox)
    topSeries    = dynamicBox.get_top()
    bottomSeries = dynamicBox.get_bottom()
    box.set_right(dynamicBox, bar_index)
    if close > topSeries or close < bottomSeries
        breakoutOccurred := true
// =========== 5)  突破條件與訊號觸發============
if boxState == "Active" and not na(dynamicBox) and not signalTriggered
    topSeries    = dynamicBox.get_top()
    bottomSeries = dynamicBox.get_bottom()

    var string buyConditionsText = ""
    var string sellConditionsText = ""

    if close > topSeries
        if buyCondition
            buyConditionsText := ""
            if adx_condition_buy
                buyConditionsText := buyConditionsText + "ADX "
            if rsi_condition_buy
                buyConditionsText := buyConditionsText + "RSI "
            if volume_condition
                buyConditionsText := buyConditionsText + "VOL "

            if not na(lastBuyBox)
                box.delete(lastBuyBox)
            lastBuyBox := box.new(left = dynamicBox.get_left(), top = topSeries, right = bar_index, bottom = bottomSeries, border_color = color.new(color.green, 0), bgcolor = buySignalColor)
            label.delete(buySignalLabel)
            buySignalLabel := label.new(bar_index, low, style = label.style_label_up, text = 'BUY\n' + buyConditionsText, color = buySignalColor, textcolor = color.white)
            signalTriggered := true
            boxState := "BuySignal"
            alert('Alert - BUY signal detected! Conditions: ' + buyConditionsText, alert.freq_once_per_bar)

            if useExitSignals
                tp1 = close + tp_multiplier_1 * priceVolatility
                tp2 = close + tp_multiplier_2 * priceVolatility
                tp3 = close + tp_multiplier_3 * priceVolatility
                sl  = low - sl_multiplier * priceVolatility

                tp_line1 := line.new(bar_index + tpOffset, tp1, bar_index + tpOffset + tpLineLength, tp1, color = color.green, style = line.style_dashed)
                tp_line2 := line.new(bar_index + tpOffset, tp2, bar_index + tpOffset + tpLineLength, tp2, color = color.green, style = line.style_dashed)
                tp_line3 := line.new(bar_index + tpOffset, tp3, bar_index + tpOffset + tpLineLength, tp3, color = color.green, style = line.style_dashed)
                sl_line  := line.new(bar_index + tpOffset, sl,  bar_index + tpOffset + tpLineLength, sl, color = color.red, style = line.style_dashed)

                tp_label1 := label.new(bar_index + tpOffset + tpLineLength, tp1, 'TP1: ' + str.tostring(tp1, format.mintick), style = label.style_label_left, color = color.green, textcolor = color.white)
                tp_label2 := label.new(bar_index + tpOffset + tpLineLength, tp2, 'TP2: ' + str.tostring(tp2, format.mintick), style = label.style_label_left, color = color.green, textcolor = color.white)
                tp_label3 := label.new(bar_index + tpOffset + tpLineLength, tp3, 'TP3: ' + str.tostring(tp3, format.mintick), style = label.style_label_left, color = color.green, textcolor = color.white)
                sl_label  := label.new(bar_index + tpOffset + tpLineLength, sl,  'SL:  ' + str.tostring(sl, format.mintick), style = label.style_label_left, color = color.red, textcolor = color.white)
        else
            boxState := "BreakoutWithoutSignal"

    else if close < bottomSeries
        if sellCondition
            sellConditionsText := ""
            if adx_condition_sell
                sellConditionsText := sellConditionsText + "ADX "
            if rsi_condition_sell
                sellConditionsText := sellConditionsText + "RSI "
            if volume_condition
                sellConditionsText := sellConditionsText + "VOL "

            if not na(lastSellBox)
                box.delete(lastSellBox)
            lastSellBox := box.new(left = dynamicBox.get_left(), top = topSeries, right = bar_index, bottom = bottomSeries, border_color = color.new(color.red, 0), bgcolor = sellSignalColor)
            label.delete(sellSignalLabel)
            sellSignalLabel := label.new(bar_index, high, style = label.style_label_down, text = 'SELL\n' + sellConditionsText, color = sellSignalColor, textcolor = color.white)
            signalTriggered := true
            boxState := "SellSignal"
            alert('Alert - SELL signal detected! Conditions: ' + sellConditionsText, alert.freq_once_per_bar)

            if useExitSignals
                tp1 = close - tp_multiplier_1 * priceVolatility
                tp2 = close - tp_multiplier_2 * priceVolatility
                tp3 = close - tp_multiplier_3 * priceVolatility
                sl  = high + sl_multiplier * priceVolatility

                tp_line1 := line.new(bar_index + tpOffset, tp1, bar_index + tpOffset + tpLineLength, tp1, color = color.green, style = line.style_dashed)
                tp_line2 := line.new(bar_index + tpOffset, tp2, bar_index + tpOffset + tpLineLength, tp2, color = color.green, style = line.style_dashed)
                tp_line3 := line.new(bar_index + tpOffset, tp3, bar_index + tpOffset + tpLineLength, tp3, color = color.green, style = line.style_dashed)
                sl_line  := line.new(bar_index + tpOffset, sl,  bar_index + tpOffset + tpLineLength, sl, color = color.red, style = line.style_dashed)

                tp_label1 := label.new(bar_index + tpOffset + tpLineLength, tp1, 'TP1: ' + str.tostring(tp1, format.mintick), style = label.style_label_left, color = color.green, textcolor = color.white)
                tp_label2 := label.new(bar_index + tpOffset + tpLineLength, tp2, 'TP2: ' + str.tostring(tp2, format.mintick), style = label.style_label_left, color = color.green, textcolor = color.white)
                tp_label3 := label.new(bar_index + tpOffset + tpLineLength, tp3, 'TP3: ' + str.tostring(tp3, format.mintick), style = label.style_label_left, color = color.green, textcolor = color.white)
                sl_label  := label.new(bar_index + tpOffset + tpLineLength, sl,  'SL:  ' + str.tostring(sl, format.mintick), style = label.style_label_left, color = color.red, textcolor = color.white)
        else
            boxState := "BreakoutWithoutSignal"



// 🚀 定義變數來追蹤訊號狀態
var bool prevBuySignal = false
var bool prevSellSignal = false

// 🚀 只有當標籤「剛出現」的 K 棒才觸發快訊
newBuySignal = not na(buySignalLabel) and not prevBuySignal and bar_index != bar_index[1]
newSellSignal = not na(sellSignalLabel) and not prevSellSignal and bar_index != bar_index[1]

// 🚀 設定 TradingView 快訊（Alert）
alertcondition(newBuySignal, title="📢 Buy Signal", message="🚀 做多motherfucker！")
alertcondition(newSellSignal, title="📢 Sell Signal", message="🔻 做空motherfucker！")

// 🚀 更新狀態變數，確保快訊不會重複觸發
prevBuySignal := newBuySignal ? true : prevBuySignal
prevSellSignal := newSellSignal ? true : prevSellSignal

// 🚀 如果沒有訊號，則重置變數（確保新訊號能再次觸發）
if na(buySignalLabel) and na(sellSignalLabel)
    prevBuySignal := false
    prevSellSignal := false
